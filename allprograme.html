public class StringManipulation
{
   
   public static String removeCharacters(String input)
   {
      if (input.length() >= 8)
      {
         // Create a new string without the 3rd and 8th characters
         return input.substring(0, 2) + input.substring(3, 7) + input.substring(8);
      }
      else if (input.length() >= 3)
      {
         // If the string is at least 3 characters long but less than 8, only remove the 3rd character
         return input.substring(0, 2) + input.substring(3);
      }
      else
      {
         // If the string is less than 3 characters long, return the original string
         return input;
      }
   }
   
   public static void main(String[] args)
   {
      String str1 = "abcdefghij";
      String result1 = removeCharacters(str1);
      System.out.println("Original string: "+ str1);
      System.out.println("Result after removing characters: "+ result1);
      
      String str2 = "abc";
      String result2 = removeCharacters(str2);
      System.out.println("Original string: "+ str2);
      System.out.println("Result after removing characters: "+ result2);
      
      String str3 = "ab";
      String result3 = removeCharacters(str3);
      System.out.println("Original string: "+ str3);
      System.out.println("Result after removing characters: "+ result3);
   }
}

public class ChineseFlag extends JPanel
{
   // Set the width and height of the flag
   private static final int FLAG_WIDTH = 800;
   private static final int FLAG_HEIGHT = 480;
   // Main method to set up the window
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Chinese National Flag");
      ChineseFlag panel = new ChineseFlag();
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      frame.setSize(FLAG_WIDTH, FLAG_HEIGHT);
      frame.add(panel);
      frame.setVisible(true);
   }
   // Paint method to draw the flag
   @Override
   protected void paintComponent(Graphics g)
   {
      super.paintComponent(g);
      // Get the graphics object
      Graphics2D g2d = (Graphics2D) g;
      // Anti-aliasing for smoother lines
      g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
      // Set the background to red (the flag's background color)
      g2d.setColor(new Color(222, 41, 19));
      // Red color of the flag
      g2d.fillRect(0, 0, FLAG_WIDTH, FLAG_HEIGHT);
      // Draw the large star
      g2d.setColor(Color.YELLOW);
      drawStar(g2d, 100, 80, 50, 20, 5);
      // Draw the smaller stars
      drawSmallStars(g2d);
   }
   // Method to draw a star at a given position
   private void drawStar(Graphics2D g2d, int x, int y, int radius1, int radius2, int numPoints)
   {
      // The angle between each point on the star
      double angle = Math.PI / numPoints;
      // Create a polygon representing the star
      GeneralPath path = new GeneralPath();
      // First point of the star
      path.moveTo(x + Math.cos(0) * radius1, y - Math.sin(0) * radius1);
      // Alternate between outer and inner points
      for (int i = 1; i < 2 * numPoints; i++)
      {
         double currentAngle = i * angle;
         double currentRadius = (i % 2 == 0) ? radius1 : radius2;
         path.lineTo(x + Math.cos(currentAngle) * currentRadius, y - Math.sin(currentAngle) * currentRadius);
      }
      path.closePath();
      g2d.fill(path);
   }
   // Draw the smaller stars on the flag
   private void drawSmallStars(Graphics2D g2d)
   {
      int[][] positions =
      {
         {
            200, 50
         }
         ,
         {
            250, 100
         }
         ,
         {
            250, 150
         }
         ,
         {
            200, 200
         }
         // Relative positions for the smaller stars
         
      };
      for (int[] position : positions)
      {
         drawStar(g2d, position[0], position[1], 20, 8, 5);
      }
   }
}

class FlagStarDrawing extends JPanel
{
   // 绘制星星的方法
   public static void paintStar(int[][] vertaxCoordinates, Graphics g)
   {
      int numVertices = vertaxCoordinates.length;
      if (numVertices < 3)
      {
         return;
      }
      
      // 定义用于存储外顶点和内顶点的x、y坐标数组
      int[] xPointsOuter = new int[5];
      int[] yPointsOuter = new int[5];
      int[] xPointsInner = new int[5];
      int[] yPointsInner = new int[5];
      
      // 提取外顶点坐标
      for (int i = 0; i < 5; i++)
      {
         xPointsOuter[i] = vertaxCoordinates[i][0];
         yPointsOuter[i] = vertaxCoordinates[i][1];
      }
      
      // 提取内顶点坐标
      for (int i = 0; i < 5; i++)
      {
         xPointsInner[i] = vertaxCoordinates[i + 5][0];
         yPointsInner[i] = vertaxCoordinates[i + 5][1];
      }
      
      // 创建一个合并外顶点和内顶点的x坐标数组
      int[] xPointsAll = new int[10];
      System.arraycopy(xPointsOuter, 0, xPointsAll, 0, 5);
      System.arraycopy(xPointsInner, 0, xPointsAll, 5, 5);
      
      // 创建一个合并外顶点和内顶点的y坐标数组
      int[] yPointsAll = new int[10];
      System.arraycopy(yPointsOuter, 0, yPointsAll, 0, 5);
      System.arraycopy(yPointsInner, 0, yPointsAll, 5, 5);
      
      // 绘制实心五角星
      g.fillPolygon(xPointsAll, yPointsAll, 10);
   }
   
   @Override
   protected void paintComponent(Graphics g)
   {
      super.paintComponent(g);
      
      // 获取大星坐标并绘制
      int[][] largeStarCoordinates = FlagStarCoordinates.starCordinate(true, 0);
      g.setColor(Color.YELLOW);
      paintStar(largeStarCoordinates, g);
      
      // 获取四颗小星坐标并依次绘制
      int[][] smallStar1Coordinates = FlagStarCoordinates.starCordinate(false, 1);
      int[][] smallStar2Coordinates = FlagStarCoordinates.starCordinate(false, 2);
      int[][] smallStar3Coordinates = FlagStarCoordinates.starCordinate(false, 3);
      int[][] smallStar4Coordinates = FlagStarCoordinates.starCordinate(false, 4);
      
      g.setColor(Color.YELLOW);
      paintStar(smallStar1Coordinates, g);
      paintStar(smallStar2Coordinates, g);
      paintStar(smallStar3Coordinates, g);
      paintStar(smallStar4Coordinates, g);
   }
   
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("五星红旗星星绘制");
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      frame.setSize(1200, 800);
      
      FlagStarDrawing panel = new FlagStarDrawing();
      frame.add(panel);
      
      frame.setVisible(true);
   }
}

class FlagStarCoordinates
{
   // 计算五星红旗上星星顶点坐标的方法
   public static int[][] starCordinate(boolean largeone, int starOrder)
   {
      int width = 1152;
      int height = width * 2 / 3;
      
      // 大星外接圆直径（基于旗高）
      int largeDiameter = height * 3 / 10;
      // 小星外接圆直径（基于旗高）
      int smallDiameter = height * 1 / 10;
      
      // 大星圆心坐标（基于旗面的相对位置计算）
      int largeCenterX = width * 5 / 10;
      int largeCenterY = height * 5 / 10;
      
      // 小星圆心坐标数组（4颗小星的圆心坐标）
      int[][] smallCenters =
      {
         {
            width * 10 / 10, height * 2 / 10
         }
         ,
         {
            width * 12 / 10, height * 4 / 10
         }
         ,
         {
            width * 12 / 10, height * 7 / 10
         }
         ,
         {
            width * 10 / 10, height * 9 / 10
         }
      };
      
      int[][] coordinates = new int[10][2];
      if (largeone)
      {
         // 计算大星的顶点坐标
         int radius = largeDiameter / 2;
         double angleStep = 2 * Math.PI / 5;
         for (int i = 0; i < 5; i++)
         {
            double angle = i * angleStep;
            // 外顶点x坐标
            coordinates[i][0] = (int) (largeCenterX + radius * Math.cos(angle));
            // 外顶点y坐标
            coordinates[i][1] = (int) (largeCenterY - radius * Math.sin(angle));
            
            double innerAngle = angle + angleStep / 2;
            // 内顶点x坐标
            coordinates[i + 5][0] = (int) (largeCenterX + radius / 2 * Math.cos(innerAngle));
            // 内顶点y坐标
            coordinates[i + 5][1] = (int) (largeCenterY - radius / 2 * Math.sin(innerAngle));
         }
      }
      else
      {
         if (starOrder >= 1 && starOrder <= 4)
         {
            // 计算对应序号小星的顶点坐标
            int radius = smallDiameter / 2;
            int centerX = smallCenters[starOrder - 1][0];
            int centerY = smallCenters[starOrder - 1][1];
            double angleStep = 2 * Math.PI / 5;
            for (int i = 0; i < 5; i++)
            {
               double angle = i * angleStep;
               // 外顶点x坐标
               coordinates[i][0] = (int) (centerX + radius * Math.cos(angle));
               // 外顶点y坐标
               coordinates[i][1] = (int) (centerY - radius * Math.sin(angle));
               
               double innerAngle = angle + angleStep / 2;
               // 内顶点x坐标
               coordinates[i + 5][0] = (int) (centerX + radius / 2 * Math.cos(innerAngle));
               // 内顶点y坐标
               coordinates[i + 5][1] = (int) (centerY - radius / 2 * Math.sin(innerAngle));
            }
         }
      }
      return coordinates;
   }
   
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("五星红旗星星绘制");
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      frame.setSize(1200, 800);
      
      FlagStarDrawing panel = new FlagStarDrawing();
      frame.add(panel);
      
      frame.setVisible(true);
      // 手动触发重绘操作
      panel.repaint();
   }
   
}
